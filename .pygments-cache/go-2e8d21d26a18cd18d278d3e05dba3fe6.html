func sum(a int, b int) int{
	return a+b
}
//也可以写成下面这样：
func sum(a,b int) int{
	return a+b
}
//函数可以有不定个数的参数

func sum(values ...int){
	var total int = 0
	for _,v :=range values{ 
				//这里用了一个变量 _，golang中如果一个变量声明了但没有使用是不允许的，所以有了这个特殊的_
				//对于_，golang在编译的时候不会做0使用的验证。如果需要访问下标，可以写作for idx,value :=range values....
		total+=v
	}
	return total
}
